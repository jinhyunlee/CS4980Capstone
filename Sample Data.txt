Question 1: 

Write a method called 'five' with no parameter that returns an int "5" 

ANSWER 1: 

#include <stdio.h>

int main() {
       printf("Five in number is %d", five());
       if (five() != 5) {
           printf("Failed");
       }
       else {
           printf("Success");
       }
       return 0;
}

CODE 1:

int five() {
	return 5;
}
